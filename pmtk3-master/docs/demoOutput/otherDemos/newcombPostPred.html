function doPlot(D)
n  = length(D);

for i=1:n
    ndx = setdiff(1:n, i);
    mu = mean(D(ndx)); sigma = std(D(ndx));
    pred(i) = normcdf(D(i), mu, sigma); % cdf not pdf
end

unif = rand(n,1);
figure;
qqplot(pred, unif);
xlabel('quantiles of uniform')
ylabel('quantiles of predictive distribution')

end
