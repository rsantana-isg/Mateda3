function ev = paretoEvidence(D, b, K)

m = max(D);
N = length(D);
if m <= b
    ev =  K/( (N+K) * b^N );
else
    ev = (K * b^K)/( (N+K).*m.^(N+K));
end


function ev = paretoEvidence2(m, N, b, K)
