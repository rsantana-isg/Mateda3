% pval = 0.1825 - retain the null at 5% level
O = [9 43; 4 44];
N=sum(sum(O));
nidot=sum(O,2);
ndotj=sum(O,1);
pdotj = ndotj/N;
pidot = nidot/N;
E=N*pidot*pdotj;
chi2 =sum(sum(((O-E).^2)./E))

[J K] = size(O);
dof = (J-1)*(K-1);
pval = 1-chi2cdf(chi2, dof)

    1.7774

    0.1825
