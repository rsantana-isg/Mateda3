Sample data

truth.mu    = 5;
truth.Sigma = 10;
setSeed(1);
X = colvec(gaussSample(truth, 100));
xyrange = [-5 15 eps 15];
figure; hold on;
[styles, colors, symbols] =  plotColors();

%
ns = [5 25 100];
str = {};
ncontours = 3;
for i=1:length(ns)

    n    = ns(i);
    data = X(1:n);
    xbar =  mean(data);
    Xc   = data-xbar;
    XX   = (Xc'*Xc)/n;

Update

    post.k     = k0 + n;
    post.dof   = v0 + n;
    post.Sigma = (S0 + n*XX + (k0*n)/(k0+n)*(xbar-m0)*(xbar-m0)');
    post.mu    = (k0*m0 + n*xbar)/post.k;
